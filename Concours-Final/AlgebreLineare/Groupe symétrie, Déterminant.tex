\subsubsection{Morphisme de groupe} % (fold)
\begin{Prop}{La signature est un morphisme de groupes}{}
\begin{tcolorbox}
    Morphisme de groups : 
    \begin{equation}
      \varepsilon( \sigma_1 \circ \sigma_2) = \varepsilon(\sigma_1) \varepsilon(\sigma_2)
    \end{equation}
\end{tcolorbox}
\end{Prop}
